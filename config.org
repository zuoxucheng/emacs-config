(defun my-minimal-header-line ()
  "Render a custom header line, manually calculating spaces to defeat margins."
  (let* ((left (format-mode-line
                '(" "
                  ;; Buffer modified status
                  mode-line-modified "  "
                  ;; Buffer name
                  (:propertize "%b" face mode-line-buffer-id) "  "
                  ;; Buffer size (restored)
                  "%I  "
                  ;; Line and column
                  "%l:%o%%"
                  )))
         (right (format-mode-line
                 '((:eval (when (bound-and-true-p org-timer-mode-line-string)
                            (concat org-timer-mode-line-string " · ")))
                   (:eval (when (and (bound-and-true-p battery-mode-line-string)
                                     (not (string-empty-p battery-mode-line-string)))
                            (concat battery-mode-line-string " · ")))
                   ;; Time string (trimming leading spaces)
                   (:eval (string-trim (or display-time-string "")))
                   " ")))
         (total-width (window-total-width))
         (left-width (string-width left))
         (right-width (string-width right))
         (space-width (max 0 (- total-width left-width right-width)))
         (padding (make-string space-width ?\s)))
    (concat left padding right)))

The solution without touching the package's source code:
Load Gruvbox without enabling it, replace the problematic and customized face definitions while they are inert, and only then enable the theme. This avoids the Emacs 31 Gnus face-inheritance cycle.
#+begin_src emacs-lisp
(use-package gruvbox-theme
  :ensure t
  :config
  (defconst my/gruvbox-palettes
    '((gruvbox-dark-medium
       :foreground "#ebdbb2"
       :blue       "#83a598"
       :muted      "#7c6f64"
       :orange     "#fe8019")
      (gruvbox-light-medium
       :foreground "#3c3836"
       :blue       "#076678"
       :muted      "#a89984"
       :orange     "#af3a03")))

  (defun my/gruvbox-apply-overrides (theme)
    "Correct and customize Gruvbox THEME before enabling it."
    (let* ((palette (cdr (assq theme my/gruvbox-palettes)))
           (foreground (plist-get palette :foreground))
           (blue (plist-get palette :blue))
           (muted (plist-get palette :muted))
           (orange (plist-get palette :orange))
           (faces
            `((gnus-group-news-low
               ((t (:weight bold :foreground ,muted))))
              (gnus-group-news-low-empty
               ((t (:weight normal :foreground ,muted))))

              (org-level-1
               ((t (:foreground ,foreground :height 1.2 :weight bold))))
              (org-level-2
               ((t (:foreground ,foreground :weight bold :underline t))))
              (org-level-3 ((t (:foreground ,blue))))
              (org-level-4 ((t (:foreground ,blue))))
              (org-level-5 ((t (:foreground ,blue))))
              (org-level-6 ((t (:foreground ,blue))))
              (org-level-7 ((t (:foreground ,blue))))
              (org-level-8 ((t (:foreground ,blue))))

              (line-number
               ((t (:inherit default
                    :foreground ,muted
                    :background unspecified))))
              (line-number-current-line
               ((t (:inherit hl-line
                    :foreground ,orange
                    :background unspecified))))

              (nobreak-space
               ((t (:foreground unspecified
                    :background unspecified
                    :underline nil)))))))
      ;; Autothemer can leave more than one setting for a face.  Remove
      ;; every old Gruvbox entry first, otherwise a stale entry can win
      ;; when `enable-theme' processes the theme settings.
      (dolist (face (mapcar #'car faces))
        (put face 'theme-face
             (assq-delete-all theme (get face 'theme-face))))
      (put theme 'theme-settings
           (seq-remove
            (lambda (setting)
              (and (eq (car setting) 'theme-face)
                   (memq (cadr setting) (mapcar #'car faces))))
            (get theme 'theme-settings)))
      (apply #'custom-theme-set-faces theme faces)))

  (defun my/load-gruvbox-theme (theme)
    "Load, correct, and enable Gruvbox THEME."
    (mapc #'disable-theme (copy-sequence custom-enabled-themes))
    (load-theme theme t t)
    (my/gruvbox-apply-overrides theme)
    (enable-theme theme))

  (defun my/toggle-gruvbox-theme ()
    "Toggle between medium dark and light Gruvbox."
    (interactive)
    (my/load-gruvbox-theme
     (if (custom-theme-enabled-p 'gruvbox-dark-medium)
         'gruvbox-light-medium
       'gruvbox-dark-medium)))

  (global-set-key (kbd "C-c t") #'my/toggle-gruvbox-theme)
  (my/load-gruvbox-theme 'gruvbox-dark-medium)

(defvar my/gptel-translate-system-prompt
  "Respond concisely. Translate the user's text naturally and accurately.
If the input is Japanese, translate it to English and Chinese.
If the input is English, translate it to Japanese and Chinese.
Keep technical terms precise. If text is a single word, provide dictionary-like information and give example sentence.")

(defun my/gptel-translate-buffer ()
  "Translate region or word at point in a dedicated gptel buffer."
  (interactive)
  (require 'gptel)
  (let ((text
         (if (use-region-p)
             (buffer-substring-no-properties (region-beginning) (region-end))
           (thing-at-point 'word t))))
    (unless text
      (user-error "No region or word at point"))
    (let ((buf (gptel my/gptel-translate-buffer-name nil nil nil)))
      (with-current-buffer buf
        (goto-char (point-max))
        (unless (bobp)
          (insert "\n\n"))
        (insert text)
        (goto-char (point-max))
        (let ((gptel--system-message my/gptel-translate-system-prompt)
              (gptel-use-context nil)
              (gptel-use-tools nil))
          (gptel-send nil)))
      (pop-to-buffer buf))))
